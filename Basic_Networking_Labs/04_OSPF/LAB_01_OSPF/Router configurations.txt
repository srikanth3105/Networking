R1_Config

hostname R1
interface g0/0/1
 ip address 192.168.1.100 255.255.255.0
 no shutdown
interface S0/1/0
 ip address 10.0.0.1 255.255.255.252
 no shutdown

# Configure OSPF
router ospf 1
 network 192.168.1.0 0.0.0.255 area 0
 network 10.0.0.0 0.0.0.3 area 0
